# For Linux

pyinstaller --onefile --clean --noconfirm --hidden-import=sip \
  --add-data "src/in:in" --add-data "data:data" -n gaitscope ./launcher.py

pyinstaller --onefile --clean --noconfirm --hidden-import=PyQt6.QtCore --add-data "src/in:in" --add-data "data:data" --icon=assets/Logo.png -n gaitscope ./launcher.py


# For Windows

pyinstaller --onedir --clean --noconfirm --hidden-import=PyQt6.QtCore --hidden-import=PyQt6.QtGui --hidden-import=PyQt6.QtWidgets --hidden-import=PyQt6.QtMultimedia --add-data "src/in;in" --add-data "data;data" --windowed -n gaitscope launcher.py
